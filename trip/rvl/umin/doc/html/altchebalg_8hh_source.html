00057   void ChebCoeff(typename ScalarFieldTraits<Scalar>::AbsType alpha, 
00058          typename ScalarFieldTraits<Scalar>::AbsType epsilon, 
00059          typename ScalarFieldTraits<Scalar>::AbsType gamma, 
00060          int & kmax,
00061          std::vector<typename ScalarFieldTraits<Scalar>::AbsType> & coeff) {
00062 
00063     typedef typename ScalarFieldTraits<Scalar>::AbsType atype;
00064     // NOTE: estimating the necessary number of iterations
00065     atype one = ScalarFieldTraits<atype>::One();
00066     atype beta=0;
00067     atype epsest=0;
00068     ProtectedDivision<atype>(sqrt(alpha)*epsilon,one+sqrt(alpha),epsest);
00069     atype epsk;
00070     epsk= epsest + one;
00071     atype tmp = one;
00072     ProtectedDivision<atype>(one - gamma,one + gamma,beta);
00073     atype q=0;
00074     ProtectedDivision<atype>(beta,one + sqrt(one-beta*beta),q);
00075     int k = 1;
00076     while (epsk > epsest && k<=kmax) {
00077       tmp = tmp * q;
00078       ProtectedDivision<atype>(2*tmp,one + tmp*tmp,epsk);
00079       k = k + 1;
00080     }
00081     kmax = k-1;
00082     // NOTE: compute Chebyshev coefficients
00083     atype ckm = one;
00084     atype ck = one / beta;
00085     atype ckp=0;
00086     coeff.reserve(kmax+1);   // allocate memory for coeff
00087     coeff[0] = ScalarFieldTraits<atype>::Zero();
00088     coeff[1] = one;
00089     for (int i=2; i<kmax+1; i++) {
00090       ProtectedDivision<atype>(2*ck,beta,ckp);
00091       ckp = ckp - ckm;
00092       coeff[i] = one + ckm / ckp;
00093       ckm = ck;
00094       ck = ckp;
00095     }
00096   }
