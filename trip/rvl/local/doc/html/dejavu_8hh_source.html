00054   template<class T>
00055   void dejavu(size_t * i, std::vector<T *> vec) {
00056     if (*i>vec.size()-1) { 
00057       RVLException e;
00058       e<<"Error: function dejavu\n";
00059       e<<"input int "<<*i<<" out of range [0, "<<vec.size()-1<<"].\n";
00060       throw e;
00061     }
00062     // save init value of *i as loop limit
00063     size_t jsave = *i;
00064     // in loop, reset *i if vec value is aliased to previous, AND
00065     // if it's still = saved value - the second condn prevents repeated
00066     // assignments
00067     for (size_t j=0;j<jsave;j++) { 
00068       if ((vec[j] == vec[*i]) &&
00069       (*i == jsave)) { *i=j; }
00070     }
00071   }
