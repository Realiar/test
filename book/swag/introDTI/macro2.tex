\def\GEOMlaw{
\strain = \frac{1}{2} 
\lb \GRAD{\uu} + \lp \GRAD{\uu} \rp^T \rb
}

\def\HOOKElaw{
\stress = \lambda \; tr \lp \strain \rp {\bf I} + 2 \mu \strain 
}

\def\NEWTONlaw{
\rho \ddot{\uu} = \DIV{\stress}
}
